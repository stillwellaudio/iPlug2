291  }
292 
293  xarray[arcpoints] = cx;
294  yarray[arcpoints] = cy;
295 
296  FillConvexPolygon(color, xarray, yarray, arcpoints + 1, pBlend);

307 bool IGraphicsLice::DoDrawMeasureText(const IText& text, const char* str, IRECT& bounds, const IBlend* pBlend, bool measure)
308 {
309  const int ds = GetScreenScale();
310  if (!str || str[0] == '\0')
311  {
312  return true;
313  }
314 
315  LICE_IFont* font = text.mCached;
316  bounds.Translate(-mDrawOffsetX, -mDrawOffsetY);
317 
318  if (!font || text.mCachedScale != ds)
319  {
320  font = CacheFont(text, ds);
321  if (!font) return false;
322  }
323 
324  LICE_pixel color;
325 
326  if (GetTextEntryControl() && GetTextEntryControl()->GetRECT() == bounds)
327  color = LiceColor(text.mTextEntryFGColor);
328  else
329  color = LiceColor(text.mFGColor);
330 
331  font->SetTextColor(color);
332 
333  UINT fmt = DT_NOCLIP;
334  if (LICE_GETA(color) < 255) fmt |= LICE_DT_USEFGALPHA;
335  if (text.mAlign == IText::kAlignNear)
336  fmt |= DT_LEFT;
337  else if (text.mAlign == IText::kAlignCenter)
338  fmt |= DT_CENTER;
339  else // if (text.mAlign == IText::kAlignFar)
340  fmt |= DT_RIGHT;
341 
342  if (text.mVAlign == IText::kVAlignTop)
343  fmt |= DT_TOP;
344  else if (text.mVAlign == IText::kVAlignBottom)
345  fmt |= DT_BOTTOM;
346  else if (text.mVAlign == IText::kVAlignMiddle)
347  fmt |= DT_VCENTER;
348 
349  if (measure)
350  {
351  fmt |= DT_CALCRECT;
352  RECT R = {0,0,0,0};
353 #if defined OS_MAC || defined OS_LINUX
354  font->SWELL_DrawText(mRenderBitmap, str, -1, &R, fmt);
355 #elif defined OS_WIN
356  font->DrawTextA(mRenderBitmap, str, -1, &R, fmt);
357 #else
358  #error NOT IMPLEMENTED
359 #endif
360  if( text.mAlign == IText::kAlignNear)
361  bounds.R = R.right;
362  else if (text.mAlign == IText::kAlignCenter)
363  {
364  bounds.L = (int) bounds.MW() - (R.right/2);
365  bounds.R = bounds.L + R.right;
366  }
367  else // (text.mAlign == IText::kAlignFar)
368  {
369  bounds.L = bounds.R - R.right;
370  bounds.R = bounds.L + R.right;
371  }
372 
373  bounds.B = bounds.T + R.bottom;
374 
375  bounds.Scale(1.0 / ds);
376  }
377  else
378  {
379  IRECT r = bounds;
380  r.Scale(ds);
381  RECT R = { (LONG) r.L, (LONG) r.T, (LONG) r.R, (LONG) r.B };
382 #if defined OS_MAC || defined OS_LINUX
383  font->SWELL_DrawText(mRenderBitmap, str, -1, &R, fmt);
384 #elif defined OS_WIN
385  font->DrawTextA(mRenderBitmap, str, -1, &R, fmt);
386 #else
387  #error NOT IMPLEMENTED
388 #endif
389  }
390 
391  return true;
392 }

403 LICE_IFont* IGraphicsLice::CacheFont(const IText& text, double scale)
404 {
405  WDL_String hashStr(text.mFont);
406  hashStr.AppendFormatted(50, "-%d-%d-%d", text.mSize, text.mOrientation, text.mStyle);
407 
408  LICE_CachedFont* font = (LICE_CachedFont*)s_fontCache.Find(hashStr.Get(), scale);
409  if (!font)
410  {
411  font = new LICE_CachedFont;
412  int h = round(text.mSize * scale);
413  int esc = 10 * text.mOrientation;
414  int wt = (text.mStyle == IText::kStyleBold ? FW_BOLD : FW_NORMAL);
415  int it = (text.mStyle == IText::kStyleItalic ? TRUE : FALSE);
416 
417  int q;
418  if (text.mQuality == IText::kQualityDefault)
419  q = DEFAULT_QUALITY;
420 #ifdef CLEARTYPE_QUALITY
421  else if (text.mQuality == IText::kQualityClearType)
422  q = CLEARTYPE_QUALITY;
423  else if (text.mQuality == IText::kQualityAntiAliased)
424 #else
425  else if (text.mQuality != IText::kQualityNonAntiAliased)
426 #endif
427  q = ANTIALIASED_QUALITY;
428  else // if (text.mQuality == IText::kQualityNonAntiAliased)
429  q = NONANTIALIASED_QUALITY;
430 
431 #ifdef OS_MAC
432  bool resized = false;
433  Resize:
434  if (h < 2) h = 2;
435 #endif
436  HFONT hFont = CreateFont(h, 0, esc, esc, wt, it, FALSE, FALSE, ANSI_CHARSET, OUT_DEFAULT_PRECIS, CLIP_DEFAULT_PRECIS, q, DEFAULT_PITCH, text.mFont);
437  if (!hFont)
438  {
439  delete(font);
440  return 0;
441  }
442  font->SetFromHFont(hFont, LICE_FONT_FLAG_OWNS_HFONT | LICE_FONT_FLAG_FORCE_NATIVE);
443 #ifdef OS_MAC
444  if (!resized && font->GetLineHeight() != h)
445  {
446  h = int((double)(h * h) / (double)font->GetLineHeight() + 0.5);
447  resized = true;
448  goto Resize;
449  }
450 #endif
451  s_fontCache.Add(font, hashStr.Get(), scale);
452  }
453  text.mCached = font;
454  text.mCachedScale = scale;
455  return font;
456 }

512 APIBitmap* IGraphicsLice::ScaleAPIBitmap(const APIBitmap* pBitmap, int scale)
513 {
514  int destW = (pBitmap->GetWidth() / pBitmap->GetScale()) * scale;
515  int destH = (pBitmap->GetHeight() / pBitmap->GetScale()) * scale;
516

543 void IGraphicsLice::ApplyShadowMask(ILayerPtr& layer, RawBitmapData& mask, const IShadow& shadow)
544 {
545  const APIBitmap* pBitmap = layer->GetAPIBitmap();
546  LICE_IBitmap* pLayerBitmap = pBitmap->GetBitmap();
547 
548  int stride = pLayerBitmap->getRowSpan() * sizeof(LICE_pixel);
549  int size = pLayerBitmap->getHeight() * stride;
550 
551  if (mask.GetSize() >= size)
552  {
553  int x = std::round(shadow.mXOffset * GetScreenScale());
554  int y = std::round(shadow.mYOffset * GetScreenScale());
555  int nRows = pBitmap->GetWidth() - std::abs(y);
556  int nCols = pBitmap->GetHeight() - std::abs(x);
557  LICE_pixel_chan* in = mask.Get() + (std::max(-x, 0) * 4) + (std::max(-y, 0) * stride);
558  LICE_pixel_chan* out = ((LICE_pixel_chan*) pLayerBitmap->getBits()) + (std::max(x, 0) * 4) + (std::max(y, 0) * stride);
559 
560  // Pre-multiply color components
561 
562  IColor color = shadow.mPattern.GetStop(0).mColor;
563  color.Clamp();
564  unsigned int ia = (color.A * static_cast<int>(Clip(shadow.mOpacity, 0.f, 1.f) * 255.0));
565  unsigned int ir = (color.R * ia);
566  unsigned int ig = (color.G * ia);
567  unsigned int ib = (color.B * ia);
568 
569  if (!shadow.mDrawForeground)
570  {
571  LICE_Clear(pLayerBitmap, 0);
572 
573  for (int i = 0 ; i < nRows; i++, in += stride, out += stride)
574  {
575  LICE_pixel_chan* chans = out;
576 
577  for (int j = 0 ; j < nCols; j++, chans += 4)
578  {
579  unsigned int maskAlpha = in[j * 4 + LICE_PIXEL_A];
580 
581  unsigned int A = (ia * maskAlpha) >> 16;
582  unsigned int R = (ir * maskAlpha) >> 16;
583  unsigned int G = (ig * maskAlpha) >> 16;
584  unsigned int B = (ib * maskAlpha) >> 16;
585 
586  _LICE_MakePixelNoClamp(chans, R, G, B, A);
587  }
588  }
589  }
590  else
591  {
592  for (int i = 0 ; i < nRows; i++, in += stride, out += stride)
593  {
594  LICE_pixel_chan* chans = out;
595 
596  for (int j = 0 ; j < nCols; j++, chans += 4)
597  {
598  unsigned int maskAlpha = in[j * 4 + LICE_PIXEL_A];
599  unsigned int alphaCmp = 255 - chans[LICE_PIXEL_A];
600 
601  unsigned int A = chans[LICE_PIXEL_A] + ((alphaCmp * ia * maskAlpha) >> 24);
602  unsigned int R = chans[LICE_PIXEL_R] + ((alphaCmp * ir * maskAlpha) >> 24);
603  unsigned int G = chans[LICE_PIXEL_G] + ((alphaCmp * ig * maskAlpha) >> 24);
604  unsigned int B = chans[LICE_PIXEL_B] + ((alphaCmp * ib * maskAlpha) >> 24);
605 
606  _LICE_MakePixelClamp(chans, R, G, B, A);
607  }
608  }
609  }
610  }
611 }
